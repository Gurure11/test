function doGet(e) {
  try {
    const SHEET_ID = "1SnhLTWgzJfkKNmIvtj8uwa-8x7TsnRtqkfX3deEuHIQ";
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName("Fuel Log");
    if (!sheet) throw new Error("Fuel Log sheet missing");
    const data = sheet.getDataRange().getValues();
    const headers = data[0];
    const rows = data.slice(1).map(row => {
      let obj = {};
      headers.forEach((h, i) => obj[h] = row[i]);
      return obj;
    });
    return ContentService.createTextOutput(JSON.stringify({ success: true, data: rows }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
